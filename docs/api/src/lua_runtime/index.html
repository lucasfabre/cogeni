luaruntime

import "github.com/lucasfabre/cogeni/src/lua_runtime"

Package luaruntime provides an environment for executing Lua scripts with specialized bindings for AST parsing, filesystem access, and data manipulation. It uses gopher-lua as the underlying engine and integrates an async scheduler to support concurrent operations within Lua scripts.

type LuaRuntime​

LuaRuntime is a wrapper around the Lua state that provides cogeni-specific modules.

type LuaRuntime struct {
    // L is the underlying gopher-lua state.
    L   *lua.LState

    // Output stores the results from 'write(id, content)' calls. Key is the block ID.
    Output map[string]string
    // OutputTargets maps block IDs to their destination target files and tags.
    OutputTargets map[string]OutputTarget
    // ProcessFunc is a hook called when cogeni.process() is invoked in Lua.
    // Parameters: (target_path, requestor_path).
    ProcessFunc func(string, string) error
    // WaitFunc is a hook called to ensure inter-file dependency ordering.
    // Parameters: (target_path, requestor_path).
    WaitFunc func(string, string) error
    // ReadFunc is a hook to retrieve buffered content for a file path.
    // Parameters: (path). Returns (content, found).
    ReadFunc func(string) (string, bool)
    // contains filtered or unexported fields
}

type OutputTarget​

OutputTarget defines the destination for content generated by a 'write()' call.

type OutputTarget struct {
    // Path is the absolute or relative filesystem path of the destination file.
    Path string
    // Tag is the unique identifier of the <generated> block within the destination file.
    // If empty, the entire destination file will be replaced.
    Tag string
}
